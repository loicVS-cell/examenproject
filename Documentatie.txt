Reis Class
package model;

import java.time.LocalDateTime;
import java.util.List;

/**
 * Class voor het representeren van een treinreis tussen twee stations.
 * Bevat informatie over vertrek- en aankomsttijden, de trein, het personeel en de prijzen voor verschillende klassen.
 */
public class Reis {
    private String vertrekStation;
    private String aankomstStation;
    private LocalDateTime vertrekTijd;
    private LocalDateTime aankomTijd;
    private Treinen trein;
    private List<Personeel> personeel;
    private double eersteKlassePrijs;
    private double tweedeKlassePrijs;

    /**
     * Maakt een nieuwe Reis aan met de gegeven parameters.
     *
     * @param vertrekStation       Het vertrekstation van de reis.
     * @param aankomstStation      Het aankomststation van de reis.
     * @param vertrekTijd          De vertrektijd van de reis.
     * @param aankomTijd           De aankomsttijd van de reis.
     * @param trein                De trein die de reis uitvoert.
     * @param personeel            Het personeel dat aan de reis gekoppeld is.
     * @param eersteKlassePrijs    De prijs voor eerste klasse.
     * @param tweedeKlassePrijs    De prijs voor tweede klasse.
     */
    public Reis(String vertrekStation, String aankomstStation, LocalDateTime vertrekTijd, LocalDateTime aankomTijd,
                Treinen trein, List<Personeel> personeel, double eersteKlassePrijs, double tweedeKlassePrijs) {
        this.vertrekStation = vertrekStation;
        this.aankomstStation = aankomstStation;
        this.vertrekTijd = vertrekTijd;
        this.aankomTijd = aankomTijd;
        this.trein = trein;
        this.personeel = personeel;
        this.eersteKlassePrijs = eersteKlassePrijs;
        this.tweedeKlassePrijs = tweedeKlassePrijs;
    }

    /**
     * Voegt een personeelslid toe aan de reis.
     *
     * @param p Het personeelslid dat toegevoegd moet worden.
     */
    public void addPersoneel(Personeel p) {
        this.personeel.add(p);
    }

    /**
     * Controleert of de reis voldoende personeel heeft.
     *
     * @return True als de reis minstens één bestuurder en drie stewards heeft, anders false.
     */
    public boolean heeftVoldoendePersoneel() {
        int bestuurders = 0;
        int stewards = 0;
        for (Personeel p : personeel) {
            if (p.getRol().equals("Bestuurder")) {
                bestuurders++;
            } else if (p.getRol().equals("Steward")) {
                stewards++;
            }
        }
        return bestuurders >= 1 && stewards >= 3;
    }
}

package model;

import java.time.LocalDate;

/**
 * Class voor het representeren van een passagier.
 * Erft van de Persoon class en voegt functionaliteit toe om te controleren of de passagier ouder is dan 18.
 */
public class Passagier extends Persoon {

    /**
     * Maakt een nieuwe Passagier aan met standaardwaarden.
     */
    public Passagier() {
        super("", "", "00.00.00-000.00", LocalDate.now());
    }

    /**
     * Maakt een nieuwe Passagier aan met de gegeven naam, achternaam, rijksregisternummer en geboortedatum.
     *
     * @param naam              De voornaam van de passagier.
     * @param achternaam        De achternaam van de passagier.
     * @param rijksregisternummer Het rijksregisternummer van de passagier.
     * @param geboortedatum     De geboortedatum van de passagier.
     */
    public Passagier(String naam, String achternaam, String rijksregisternummer, LocalDate geboortedatum) {
        super(naam, achternaam, rijksregisternummer, geboortedatum);
    }

    /**
     * Controleert of de passagier ouder is dan 18.
     *
     * @return True als de passagier ouder is dan 18, anders false.
     */
    public boolean IsOuderDan18() {
        return java.time.Period.between(this.getGeboortedatum(), LocalDate.now()).getYears() >= 18;
    }
}
